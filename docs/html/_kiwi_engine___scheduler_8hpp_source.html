148  template<class Clock>
149  void Scheduler<Clock>::Queue::insert(std::shared_ptr<Event> event)
150  {
151  remove(event);
152 
153  auto event_it = m_events.begin();
154 
155  while(event_it != m_events.end())
156  {
157  if (event->m_time < (*event_it)->m_time)
158  {
159  m_events.insert(event_it, std::move(event));
160  break;
161  }
162 
163  ++event_it;
164  }
165 
166  if (event_it == m_events.end())
167  {
168  m_events.emplace_back(std::move(event));
169  }

172  template<class Clock>
173  void Scheduler<Clock>::Queue::process(time_point_t process_time)
174  {
175  std::lock_guard<std::mutex> lock(m_mutex);
176 
177  size_t command_size = m_commands.load_size();
178 
179  for (size_t i = 0; i < command_size; ++i)
180  {
181  Command command;
182 
183  if (m_commands.pop(command))
184  {
185  if (command.m_time != clock_t::time_point::max())
186  {
187  command.m_event->m_time = command.m_time;
188  insert(std::move(command.m_event));
189  }
190  else
191  {
192  remove(command.m_event);
193  }
194  }
195  }
196 
197  m_events.erase(std::remove_if(m_events.begin(), m_events.end(), [&process_time](auto& event) {
198 
199  if (event->m_time <= process_time)
200  {
201  event->execute();
202  return true;
203  }
204 
205  return false;
206 
207  }), m_events.end());
208  }
