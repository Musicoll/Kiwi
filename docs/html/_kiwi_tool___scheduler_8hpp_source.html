220  {
221  size_t command_size = m_commands.load_size();
222 
223  for (size_t i = 0; i < command_size; ++i)
224  {
225  Command command;

227  if (m_commands.pop(command))

231  if (event.m_time != clock_t::time_point::max())

243

246  event.execute();
