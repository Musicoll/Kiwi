404  {
405  bool state = false;
406  std::ostringstream ss;
407  for(const auto& iter : text)

414  case '/': ss << '/'; break;
415  case 'b': ss << '\b'; break;
416  case 'f': ss << '\f'; break;
417  case 'n': ss << '\n'; break;
418  case 'r': ss << '\r'; break;
419  case 't': ss << '\t'; break;
420  case '\\': ss << '\\'; break;
421  default: ss << iter; break;
422  }
423  state = false;
424  }
425  else
426  {
427  switch(iter)
428  {
429  case '"': { return ss.str(); }
430  case '\\': { state = true; break; }

452  static std::vector<Atom> parse(std::string const& text)
453  {
454  std::vector<Atom> atoms;
455  const auto textlen = text.length();
456  auto pos = text.find_first_not_of(' ', 0);
457 
458  while(pos < textlen)
459  {
460  std::string word;
461  word.reserve(20);
462  bool is_tag = false;
463  bool is_number = false;
464  bool is_float = false;
465  bool is_negative = false;
466  bool is_quoted = false;
467 
468  while(pos < textlen)
469  {
470  const char c = text[pos];
471 
472  if(c == ' ')
473  {
474  if(!is_quoted)
475  {
476  // preserve white space in quoted tags, otherwise skip them
477  if(word.empty())
478  {
479  pos++;
480  continue;
481  }
482  else
483  {
484  break;
485  }
486  }
487  }
488  else if(c == '\"')
489  {
490  // closing quote
491  if(is_quoted)
492  {
493  pos++;
494  break;
495  }
496 
497  // opening quote
498  if(word.empty())
499  {
500  pos++;
501 
502  // ignore if it can not be closed
503  if(text.find_first_of('\"', pos) != std::string::npos)
504  is_quoted = is_tag = true;
505 
506  continue;
507  }
508  }
509  else if(!is_tag)
510  {
511  if(word.empty() && c == '-')
512  {
513  is_negative = true;
514  }
515  else if(!is_float && (word.empty() || is_number || is_negative) && c == '.')
516  {
517  is_float = true;
518  }
519  else if(isdigit(c) && (is_number || (word.empty() || is_negative || is_float)))
520  {
521  is_number = true;
522  }
523  else
524  {
525  is_tag = true;
526  is_number = is_negative = is_float = false;
527  }
528  }
529 
530  word += c;
531  pos++;
532  }
533 
534  if(!word.empty())
535  {
536  if(is_number)
537  {
538  if(is_float)
539  {
540  atoms.emplace_back(std::stod(word.c_str()));
541  }
542  else
543  {
544  long value = 0;
545  try
546  {
547  value = std::stol(word.c_str());
548  }
549  catch(std::out_of_range const& err)
550  {
551  // print error here
552  }
553  catch(std::invalid_argument const& err)
554  {
555  // print error here
556  }
557 
558  atoms.emplace_back(value);
559  }
560  }
561  else
562  {
563  atoms.emplace_back(StringHelper::unescape(word));
564  }
565  }
566  }
567 
568  return atoms;
569  }

571  static std::string trimDecimal(std::string& text)
572  {
573  std::string::size_type pos = text.find('.');
574 
575  if(pos != std::string::npos)

577  std::string::size_type size = text.size();

582  }
